Layout breakpoints
< 300px — Stacked 300-500px — Horizontal > 500px — Full 
.container {
  container-type: inline-size;
  container-name: card;
}

/* Stacked (default, < 300px) */
.card { flex-direction: column; }
.card-image { width: 100%; height: 120px; }

/* Horizontal (300px - 500px) */
@container card (min-width: 300px) {
  .card { flex-direction: row; }
  .card-image { width: 140px; height: 100px; }
}

/* Full layout (> 500px) */
@container card (min-width: 500px) {
  .card { flex-direction: row; }
  .card-image { width: 180px; height: 130px; }
  .card-meta { display: flex; }
  .card-actions { display: flex; }
}
